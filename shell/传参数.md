#1、使用说明
usage()
{
    cat << EOF
install 

Usage : exec.sh --host ${IP} --user ${user} --password ${password} --command ${command} --port ${port}
    -h | --help        : Show  message
    --host             : host ip address
    --user             : host username
    --password         : host password
    --command          : exec command

        Examples :
        ./exec.sh --host 192.168.1.66 --user root --password 123456 --command date --port 20022
EOF
}

2、获取当前路径
BASEDIR=$(dirname "$0")
cd $BASEDIR
BASEDIR=`pwd`

3、获取命令传递参数
while (($#)); do
    case "$1" in
        -h | --help)   usage;   exit 0;;
        --host)  host=${2}; shift 2;;
        --user)  user=${2}; shift 2;;
        --password)  password=${2}; shift 2;;
        --command)  command=${2}; shift 2;;
        --port)  command=${2}; shift 2;;
        *)
            usage
            echo "!!!ERROR : Unknown option"
            exit 3
        ;;
    esac
done

4、判断参数是否为空
if [ -z ${host} ]; then
     echo "!!!ERROR: host is missing!"
     exit 1
fi

if [ -z ${user} ]; then
     echo "!!!ERROR: user is missing!"
     exit 1
fi

if [ -z ${password} ];then
	echo "!!!ERROR: password is missing!"
	exit 1
fi

if [ -z ${command} ];then
	echo "!!!ERROR: command is missing!"
	exit 1
fi

if [ -z ${port} ];then
	echo "!!!ERROR: command is missing!"
	exit 1
fi

5、执行的代码块
shell(){
    ssh -p ${password} ${user}@${host} ${command} -P ${port}
}
