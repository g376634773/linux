usage()
{
    cat << EOF
install 

Usage : install.sh --host ${IP} --user ${user} --password ${password} --command ${command}
    -h | --help        : Show  message
    --host             : host ip address
    --user             : host username
    --password         : host password
    --command          : exec command

        ex :
        ./install.sh --host 192.168.1.66 --user root --password 123456 --command date
EOF
}

BASEDIR=$(dirname "$0")
cd $BASEDIR
BASEDIR=`pwd`

while (($#)); do
    case "$1" in
        -h | --help)   usage;   exit 0;;
        -n | --nas) nas=${2}; shift 2;;
        --host)  host=${2}; shift 2;;
        --user)  user=${2}; shift 2;;
        --password)  password=${2}; shift 2;;
        --command)  command=${2}; shift 2;;
        *)
            usage
            echo "!!!ERROR : Unknown option"
            exit 3
        ;;
    esac
done

if [ -z ${host} ]; then
     echo "!!!ERROR: host is missing!"
     exit 1
fi

if [ -z ${user} ]; then
     echo "!!!ERROR: user is missing!"
     exit 1
fi

if [ -z ${password} ];then
	echo "!!!ERROR: password is missing!"
	exit 1
fi

if [ -z ${command} ];then
	echo "!!!ERROR: command is missing!"
	exit 1
fi
