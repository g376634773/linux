#rename命令改名
rename "_finished" "" *.jpg

#for循环加变量部分截取
for file in `ls *.jpg`;do mv $file `echo ${file%_finished*}.jpg`;done;

#不使用echo也可以实现
for file in `ls *.jpg`;do mv $file ${file%_finished*}.jpg;done;
